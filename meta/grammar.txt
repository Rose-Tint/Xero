expr ::= expression

id ::= identifier

sym ::= symbol
      | {
      | }
      | [
      | ]
      | :
      | (
      | )
      | "
      | '
      | ;
      | ~

op ::= operator
     | @
     | %
     | ^
     | &
     | *
     | -
     | =
     | +
     | <
     | >
     | ,
     | .
     | /
     | |

struct-decl ::= "struct" id (":" "mod" (id ",")+)?

var-decl ::= mod* id ":" id

struct-def ::= struct-decl "{" expr* "}"

var-assignment ::= id "=" expr+
